287  void start_element(const XML_Char* element, const XML_Char** attrs) {
288  try {
289  switch (m_context) {
290  case context::root:
291  if (!strcmp(element, "osm") || !strcmp(element, "osmChange")) {
292  if (!strcmp(element, "osmChange")) {
293  m_header.has_multiple_object_versions(true);
294  }
295  for (int count = 0; attrs[count]; count += 2) {
296  if (!strcmp(attrs[count], "version")) {
297  if (strcmp(attrs[count+1], "0.6")) {
298  throw std::runtime_error("can only read version 0.6 files");
299  }
300  } else if (!strcmp(attrs[count], "generator")) {
301  m_header.set("generator", attrs[count+1]);
302  }
303  }
304  }
305  m_context = context::top;
306  break;
307  case context::top:
308  assert(!m_tl_builder);
309  if (!strcmp(element, "node")) {
310  if (!m_promise_fulfilled) {
311  header_is_done();
312  }
313  if (m_read_types & osmium::osm_entity::flags::node) {
314  m_node_builder = std::unique_ptr<osmium::osm::NodeBuilder>(new osmium::osm::NodeBuilder(m_buffer));
315  m_node_builder->add_user(init_object(m_node_builder->object(), attrs));
316  m_context = context::node;
317  } else {
318  m_context = context::ignored_node;
319  }
320  } else if (!strcmp(element, "way")) {
321  if (!m_promise_fulfilled) {
322  header_is_done();
323  }
324  if (m_read_types & osmium::osm_entity::flags::way) {
325  m_way_builder = std::unique_ptr<osmium::osm::WayBuilder>(new osmium::osm::WayBuilder(m_buffer));
326  m_way_builder->add_user(init_object(m_way_builder->object(), attrs));
327  m_context = context::way;
328  } else {
329  m_context = context::ignored_way;
330  }
331  } else if (!strcmp(element, "relation")) {
332  if (!m_promise_fulfilled) {
333  header_is_done();
334  }
335  if (m_read_types & osmium::osm_entity::flags::relation) {
336  m_relation_builder = std::unique_ptr<osmium::osm::RelationBuilder>(new osmium::osm::RelationBuilder(m_buffer));
337  m_relation_builder->add_user(init_object(m_relation_builder->object(), attrs));
338  m_context = context::relation;
339  } else {
340  m_context = context::ignored_relation;
341  }
342  } else if (!strcmp(element, "changeset")) {
343  if (!m_promise_fulfilled) {
344  header_is_done();
345  }
346  if (m_read_types & osmium::osm_entity::flags::changeset) {
347  m_changeset_builder = std::unique_ptr<osmium::osm::ChangesetBuilder>(new osmium::osm::ChangesetBuilder(m_buffer));
348  init_changeset(m_changeset_builder.get(), attrs);
349  m_context = context::changeset;
350  } else {
351  m_context = context::ignored_changeset;
352  }
353  } else if (!strcmp(element, "bounds")) {
354  osmium::Location min;
355  osmium::Location max;
356  for (int count = 0; attrs[count]; count += 2) {
357  if (!strcmp(attrs[count], "minlon")) {
358  min.lon(atof(attrs[count+1]));
359  } else if (!strcmp(attrs[count], "minlat")) {
360  min.lat(atof(attrs[count+1]));
361  } else if (!strcmp(attrs[count], "maxlon")) {
362  max.lon(atof(attrs[count+1]));
363  } else if (!strcmp(attrs[count], "maxlat")) {
364  max.lat(atof(attrs[count+1]));
365  }
366  }
367  osmium::Box box;
368  box.extend(min).extend(max);
369  m_header.add_box(box);
370  } else if (!strcmp(element, "delete")) {
371  m_in_delete_section = true;
372  }
373  break;
374  case context::node:
375  m_last_context = context::node;
376  m_context = context::in_object;
377  check_tag(m_node_builder.get(), element, attrs);
378  break;
379  case context::way:
380  m_last_context = context::way;
381  m_context = context::in_object;
382  if (!strcmp(element, "nd")) {
383  m_tl_builder.reset();
384 
385  if (!m_wnl_builder) {
386  m_wnl_builder = std::unique_ptr<osmium::osm::WayNodeListBuilder>(new osmium::osm::WayNodeListBuilder(m_buffer, m_way_builder.get()));
387  }
388 
389  for (int count = 0; attrs[count]; count += 2) {
390  if (!strcmp(attrs[count], "ref")) {
391  m_wnl_builder->add_way_node(osmium::string_to_object_id(attrs[count+1]));
392  }
393  }
394  } else {
395  check_tag(m_way_builder.get(), element, attrs);
396  }
397  break;
398  case context::relation:
399  m_last_context = context::relation;
400  m_context = context::in_object;
401  if (!strcmp(element, "member")) {
402  m_tl_builder.reset();
403 
404  if (!m_rml_builder) {
405  m_rml_builder = std::unique_ptr<osmium::osm::RelationMemberListBuilder>(new osmium::osm::RelationMemberListBuilder(m_buffer, m_relation_builder.get()));
406  }
407 
408  char type = 'x';
409  object_id_type ref = 0;
410  const char* role = "";
411  for (int count = 0; attrs[count]; count += 2) {
412  if (!strcmp(attrs[count], "type")) {
413  type = static_cast<char>(attrs[count+1][0]);
414  } else if (!strcmp(attrs[count], "ref")) {
415  ref = osmium::string_to_object_id(attrs[count+1]);
416  } else if (!strcmp(attrs[count], "role")) {
417  role = static_cast<const char*>(attrs[count+1]);
418  }
419  }
420  // XXX assert type, ref, role are set
421  m_rml_builder->add_member(char_to_item_type(type), ref, role);
422  } else {
423  check_tag(m_relation_builder.get(), element, attrs);
424  }
425  break;
426  case context::changeset:
427  m_last_context = context::changeset;
428  m_context = context::in_object;
429  check_tag(m_changeset_builder.get(), element, attrs);
430  break;
431  case context::ignored_node:
432  case context::ignored_way:
433  case context::ignored_relation:
434  case context::ignored_changeset:
435  break;
436  case context::in_object:
437  // fallthrough
438  default:
439  assert(false); // should never be here
440  }
441  } catch (osmium::memory::BufferIsFull&) {
442  std::cerr << "BUFFER FULL (start_element)" << std::endl;
443  exit(1);
444  }

447  void end_element(const XML_Char* element) {
448  try {
449  switch (m_context) {
450  case context::root:
451  assert(false); // should never be here
452  break;
453  case context::top:
454  if (!strcmp(element, "osm") || !strcmp(element, "osmChange")) {
455  m_context = context::root;
456  m_queue.push(std::move(m_buffer));
457  m_queue.push(osmium::memory::Buffer()); // empty buffer to signify eof
458  } else if (!strcmp(element, "delete")) {
459  m_in_delete_section = false;
460  }
461  break;
462  case context::node:
463  assert(!strcmp(element, "node"));
464  m_tl_builder.reset();
465  m_node_builder.reset();
466  m_buffer.commit();
467  m_context = context::top;
468  flush_buffer();
469  break;
470  case context::way:
471  assert(!strcmp(element, "way"));
472  m_tl_builder.reset();
473  m_wnl_builder.reset();
474  m_way_builder.reset();
475  m_buffer.commit();
476  m_context = context::top;
477  flush_buffer();
478  break;
479  case context::relation:
480  assert(!strcmp(element, "relation"));
481  m_tl_builder.reset();
482  m_rml_builder.reset();
483  m_relation_builder.reset();
484  m_buffer.commit();
485  m_context = context::top;
486  flush_buffer();
487  break;
488  case context::changeset:
489  assert(!strcmp(element, "changeset"));
490  m_tl_builder.reset();
491  m_changeset_builder.reset();
492  m_buffer.commit();
493  m_context = context::top;
494  flush_buffer();
495  break;
496  case context::in_object:
497  m_context = m_last_context;
498  break;
499  case context::ignored_node:
500  if (!strcmp(element, "node")) {
501  m_context = context::top;
502  }
503  break;
504  case context::ignored_way:
505  if (!strcmp(element, "way")) {
506  m_context = context::top;
507  }
508  break;
509  case context::ignored_relation:
510  if (!strcmp(element, "relation")) {
511  m_context = context::top;
512  }
513  break;
514  case context::ignored_changeset:
515  if (!strcmp(element, "changeset")) {
516  m_context = context::top;
517  }
518  break;
519  default:
520  assert(false); // should never be here
521  }
522  } catch (osmium::memory::BufferIsFull&) {
523  std::cerr << "BUFFER FULL (end_element)" << std::endl;
524  exit(1);
525  }
