156  void complete_relation(osmium::relations::RelationMeta& relation_meta) {
157  const osmium::Relation& relation = this->get_relation(relation_meta);
158  std::vector<size_t> offsets;
159  for (const auto& member : relation.members()) {

162  }

167  possibly_flush_output_buffer();

172  // clear member metas
173  for (const auto& member : relation.members()) {
174  if (member.ref() != 0) {
175  auto& mmv = this->member_meta(member.type());
176  auto range = std::equal_range(mmv.begin(), mmv.end(), osmium::relations::MemberMeta(member.ref()));
177  assert(range.first != range.second);
178 
179  // if this is the last time this object was needed
180  // then mark it as removed

182  this->get_member(range.first->buffer_offset()).set_removed(true);
183  }
184

187  it->remove();
