116  void parse_format(const std::string& format) {
117  std::vector<std::string> options;
118  boost::split(options, format, boost::is_any_of(","));
119 
120  // if the first item in the format list doesn't contain
121  // an equals sign, it is a format
122  if (!options.empty() && options[0].find_first_of('=') == std::string::npos) {
123  detect_format_from_suffix(options[0]);
124  options.erase(options.begin());
125  }
126 
127  for (auto& option : options) {
128  size_t pos = option.find_first_of('=');
129  if (pos == std::string::npos) {
130  set(option, true);
131  } else {
132  std::string value = option.substr(pos+1);
133  option.erase(pos);
134  set(option, value);
135  }
136  }

145  void detect_format_from_suffix(const std::string& name) {
146  std::vector<std::string> suffixes;
147  boost::split(suffixes, name, boost::is_any_of("."));
148 
149  if (suffixes.empty()) return;
150 
151  // if the last suffix is one of a known set of compressions,
152  // set that compression
153  if (suffixes.back() == "gz") {
154  m_file_compression = file_compression::gzip;
155  suffixes.pop_back();
156  } else if (suffixes.back() == "bz2") {
157  m_file_compression = file_compression::bzip2;
158  suffixes.pop_back();
159  }
160 
161  if (suffixes.empty()) return;
162 
163  // if the last suffix is one of a known set of formats,
164  // set that format
165  if (suffixes.back() == "pbf") {
166  m_file_format = file_format::pbf;
167  suffixes.pop_back();
168  } else if (suffixes.back() == "xml") {
169  m_file_format = file_format::xml;
170  suffixes.pop_back();
171  } else if (suffixes.back() == "opl") {
172  m_file_format = file_format::opl;
173  suffixes.pop_back();
174  }
175 
176  if (suffixes.empty()) return;
177 
178  if (suffixes.back() == "osm") {
179  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
180  suffixes.pop_back();
181  } else if (suffixes.back() == "osh") {
182  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
183  m_has_multiple_object_versions = true;
184  suffixes.pop_back();
185  } else if (suffixes.back() == "osc") {
186  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
187  m_has_multiple_object_versions = true;
188  set("xml_change_format", true);
189  suffixes.pop_back();
190  }
