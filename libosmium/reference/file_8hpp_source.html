142  void parse_format(const std::string& format) {
143  std::vector<std::string> options = detail::split(format, ',');
144 
145  // if the first item in the format list doesn't contain
146  // an equals sign, it is a format
147  if (!options.empty() && options[0].find_first_of('=') == std::string::npos) {
148  detect_format_from_suffix(options[0]);
149  options.erase(options.begin());
150  }
151 
152  for (auto& option : options) {
153  size_t pos = option.find_first_of('=');
154  if (pos == std::string::npos) {
155  set(option, true);

170  void detect_format_from_suffix(const std::string& name) {
171  std::vector<std::string> suffixes = detail::split(name, '.');
172 
173  if (suffixes.empty()) return;
174 
175  // if the last suffix is one of a known set of compressions,
176  // set that compression
177  if (suffixes.back() == "gz") {
178  m_file_compression = file_compression::gzip;
179  suffixes.pop_back();
180  } else if (suffixes.back() == "bz2") {
181  m_file_compression = file_compression::bzip2;
182  suffixes.pop_back();
183  }
184 
185  if (suffixes.empty()) return;
186 
187  // if the last suffix is one of a known set of formats,
188  // set that format
189  if (suffixes.back() == "pbf") {
190  m_file_format = file_format::pbf;
191  suffixes.pop_back();
192  } else if (suffixes.back() == "xml") {
193  m_file_format = file_format::xml;
194  suffixes.pop_back();
195  } else if (suffixes.back() == "opl") {
196  m_file_format = file_format::opl;
197  suffixes.pop_back();
198  }
199 
200  if (suffixes.empty()) return;
201 
202  if (suffixes.back() == "osm") {
203  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
204  suffixes.pop_back();
205  } else if (suffixes.back() == "osh") {
206  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
207  m_has_multiple_object_versions = true;
208  suffixes.pop_back();
209  } else if (suffixes.back() == "osc") {
210  if (m_file_format == file_format::unknown) m_file_format = file_format::xml;
211  m_has_multiple_object_versions = true;
