91  void append_encoded_string(const std::string& data) {
92  utf8::unchecked::iterator<std::string::const_iterator> it {data.cbegin()};
93  utf8::unchecked::iterator<std::string::const_iterator> end {data.cend()};
94 
95  for (; it != end; ++it) {
96  uint32_t c = *it;
97 
98  // This is a list of Unicode code points that we let
99  // through instead of escaping them. It is incomplete
100  // and can be extended later.
101  // Generally we don't want to let through any character
102  // that has special meaning in the OPL format such as
103  // space, comma, @, etc. and any non-printing characters.
104  if ((0x0021 <= c && c <= 0x0024) ||
105  (0x0026 <= c && c <= 0x002b) ||
106  (0x002d <= c && c <= 0x003c) ||
107  (0x003e <= c && c <= 0x003f) ||
108  (0x0041 <= c && c <= 0x007e) ||
109  (0x00a1 <= c && c <= 0x00ac) ||
110  (0x00ae <= c && c <= 0x05ff)) {
111  utf8::unchecked::append(c, std::back_inserter(m_out));
112  } else {
113  m_out += '%';
114  snprintf(m_tmp_buffer, tmp_buffer_size, "%04x", c);
115  m_out += m_tmp_buffer;
116  }
117  }
118  }

120  void write_meta(const osmium::Object& object) {
121  snprintf(m_tmp_buffer, tmp_buffer_size, "%" PRId64 " v%d d", object.id(), object.version());
122  m_out += m_tmp_buffer;
123  m_out += (object.visible() ? 'V' : 'D');
124  snprintf(m_tmp_buffer, tmp_buffer_size, " c%d t", object.changeset());
125  m_out += m_tmp_buffer;
126  m_out += object.timestamp().to_iso();
127  snprintf(m_tmp_buffer, tmp_buffer_size, " i%d u", object.uid());
128  m_out += m_tmp_buffer;
129  append_encoded_string(object.user());
130  m_out += " T";
131  bool first = true;
132  for (auto& tag : object.tags()) {
133  if (first) {
134  first = false;
135  } else {
136  m_out += ',';
137  }
138  append_encoded_string(tag.key());
139  m_out += '=';
140  append_encoded_string(tag.value());
141  }
