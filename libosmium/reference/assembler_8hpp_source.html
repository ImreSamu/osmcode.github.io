561  bool create_rings() {
562  m_segment_list.sort();
563  m_segment_list.erase_duplicate_segments();
564 
565  // Now we look for segments crossing each other. If there are
566  // any, the multipolygon is invalid.
567  // In the future this could be improved by trying to fix those
568  // cases.
569  if (m_segment_list.find_intersections(m_config.problem_reporter)) {
570  return false;
571  }
572 
573  // Now iterator over all segments and add them to rings. Each segment
574  // is tacked on to either end of an existing ring if possible, or a
575  // new ring is started with it.
576  for (const auto& segment : m_segment_list) {
577  if (debug()) {
578  std::cerr << " checking segment " << segment << "\n";
579  }
580  if (!add_to_existing_ring(segment)) {
581  if (debug()) {
582  std::cerr << " new ring for segment " << segment << "\n";
583  }
584  m_rings.emplace_back(segment);
585  }
586  }
587 
588  if (debug()) {
589  std::cerr << " Rings:\n";
590  for (const auto& ring : m_rings) {
591  std::cerr << " " << ring;
592  if (ring.closed()) {
593  std::cerr << " (closed)";
594  }
595  std::cerr << "\n";
596  }
597  }
598 
599  if (check_for_open_rings()) {
600  if (debug()) {
601  std::cerr << " not all rings are closed\n";
602  }
603  return false;
604  }
605 
606  if (debug()) {
607  std::cerr << " Find inner/outer...\n";
608  }
609 
610  if (m_rings.size() == 1) {
611  m_outer_rings.push_back(&m_rings.front());
612  } else {
613  for (auto& ring : m_rings) {
614  check_inner_outer(ring);
615  if (ring.outer()) {
616  if (!ring.is_cw()) {
617  ring.reverse();
618  }
619  m_outer_rings.push_back(&ring);
620  } else {
621  if (ring.is_cw()) {
622  ring.reverse();
623  }
624  m_inner_rings.push_back(&ring);
625  }
626  }
627 
628  if (m_outer_rings.size() == 1) {
629  for (auto inner : m_inner_rings) {
630  m_outer_rings.front()->add_inner_ring(inner);
631  }
632  } else {
633  // sort outer rings by size, smallest first
634  std::sort(m_outer_rings.begin(), m_outer_rings.end(), [](ProtoRing* a, ProtoRing* b) {
635  return a->area() < b->area();
636  });
637  for (auto inner : m_inner_rings) {
638  for (auto outer : m_outer_rings) {
639  if (inner->is_in(outer)) {
640  outer->add_inner_ring(inner);
641  break;
642  }
643  }
644  }
645  }
646  }
647 
648  check_inner_outer_roles();
649 
650  return true;
651  }
