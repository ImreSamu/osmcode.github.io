142  int execute(const std::string& command, const std::string& filename) {
143  int pipefd[2];
144  if (pipe(pipefd) < 0) {
145  throw std::system_error(errno, std::system_category(), "opening pipe failed");
146  }
147  pid_t pid = fork();
148  if (pid < 0) {
149  throw std::system_error(errno, std::system_category(), "fork failed");
150  }
151  if (pid == 0) { // child
152  // close all file descriptors except one end of the pipe
153  for (int i=0; i < 32; ++i) {
154  if (i != pipefd[1]) {
155  ::close(i);
156  }
157  }
158  if (dup2(pipefd[1], 1) < 0) { // put end of pipe as stdout/stdin
159  exit(1);
160  }
161 
162  ::open("/dev/null", O_RDONLY); // stdin
163  ::open("/dev/null", O_WRONLY); // stderr
164  if (::execlp(command.c_str(), command.c_str(), filename.c_str(), nullptr) < 0) {
165  exit(1);
166  }
167  }
168  // parent
169  m_childpid = pid;
