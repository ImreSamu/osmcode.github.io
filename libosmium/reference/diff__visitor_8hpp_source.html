158  template <class TIterator, class ...TVisitors>
159  inline void apply_diff(TIterator it, TIterator end, TVisitors&... visitors) {
160  typedef osmium::DiffIterator<TIterator> diff_iterator;
161 
162  diff_iterator dit(it, end);
163  diff_iterator dend(end, end);
164 
165  osmium::item_type last_type = osmium::item_type::undefined;
166 
167  for (; dit != dend; ++dit) {
168  if (last_type != dit->type()) {
169  osmium::diff_visitor::detail::apply_before_and_after_recurse(last_type, dit->type(), visitors...);
170  last_type = dit->type();
171  }
172  osmium::diff_visitor::detail::apply_diff_iterator_recurse(*dit, visitors...);
173  }
174 
175  osmium::diff_visitor::detail::apply_before_and_after_recurse(last_type, osmium::item_type::undefined, visitors...);
176  }
