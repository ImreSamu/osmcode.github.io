607

613  std::this_thread::sleep_for(std::chrono::milliseconds(10));
614  }
615  } else {
616  std::promise<osmium::memory::Buffer> promise;
617  m_queue.push(promise.get_future());
618  promise.set_value(data_blob_parser());
619  }
620  ++n;
621 
622  // wait if the backlog of buffers with parsed data is too large
623  while (!m_done && m_queue.size() > m_max_buffer_queue_size) {
624  std::this_thread::sleep_for(std::chrono::milliseconds(10));
625  }
626 
627  if (m_done) {
628  return;
629  }
630  }
631  m_done = true;
