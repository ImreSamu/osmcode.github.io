328  unsigned char* reserve_space(const size_t size) {
329  if (m_written + size > m_capacity) {
330  if (m_full) {
331  m_full(*this);
332  } else if (!m_memory.empty() && (m_auto_grow == auto_grow::yes)) {
333  // double buffer size until there is enough space
334  size_t new_capacity = m_capacity * 2;
335  while (m_written + size > new_capacity) {
336  new_capacity *= 2;
337  }
338  grow(new_capacity);
339  } else {
340  throw BufferIsFull();
341  }
342  }
343  unsigned char* data = &m_data[m_written];
344  m_written += size;
345  return data;
346  }
