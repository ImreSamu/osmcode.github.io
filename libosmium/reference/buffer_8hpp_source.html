307  unsigned char* reserve_space(const size_t size) {
308  if (m_written + size > m_capacity) {
309  if (m_full) {
310  m_full(*this);
311  } else if (!m_memory.empty() && m_auto_grow) {
312  // double buffer size until there is enough space
313  size_t new_capacity = m_capacity * 2;
314  while (m_written + size > new_capacity) {
315  new_capacity *= 2;
316  }
317  grow(new_capacity);
318  } else {
319  throw BufferIsFull();
320  }
321  }
322  unsigned char* data = &m_data[m_written];
323  m_written += size;
324  return data;
325  }
