79  void xml_string(std::string& out, const char* in) {
80  for (; *in != '\0'; ++in) {
81  switch(*in) {
82  case '&': out += "&amp;"; break;
83  case '\"': out += "&quot;"; break;
84  case '\'': out += "&apos;"; break;
85  case '<': out += "&lt;"; break;
86  case '>': out += "&gt;"; break;
87  default: out += *in; break;
88  }
89  }
90  }
